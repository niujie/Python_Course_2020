\begin{frame}[fragile]{Functions}
The purpose of using function (in any programming language) is - \textbf{RE-USING} it.\\
\begin{pythoncode}
>>> def f():
...     pass
...
>>> f()
>>> f
<function f at 0x123ABCDEF>
>>>
\end{pythoncode}
\end{frame}

\begin{frame}[fragile]{Functions}
\begin{pythoncode}
>>> def ping():
...     return "Ping!"
...
>>> ping()
'Ping!'
>>> x = ping()
>>> print(x)
Ping!
>>> dir()
['__builtins__', ... ..., 'f', 'ping', 'x']
>>>
\end{pythoncode}
\end{frame}
